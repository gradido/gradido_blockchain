76 template <typename FLDataType> DRMemoryList<FLDataType>::DRMemoryList(int iNumObjects)

81 
82  //Speicher reservieren
83  m_pObjectData = new FLDataType[iNumObjects];

86  //Checken ob der Speicher reserviert wurde

91 
92  //Anzahl zuweisen

95  m_bFreeOnDestroy = true;

143 template <typename FLDataType> void DRMemoryList<FLDataType>::FillStack()
144 {
145  int iIndex = m_iNumObject - 1;
146 
147  //Die Zeiger in m_ppFreeObjects werden von vorne nach hinten,
148  //von hinten nach vorne auf den reservierten Speicher ausgerichtet
149  for(m_iTop = 0; m_iTop < m_iNumObject; m_iTop++)
150  {
151  m_ppFreeObjects[m_iTop] = &(m_pObjectData[iIndex--]);
152  }
153 }

157 template <typename FLDataType> FLDataType* DRMemoryList<FLDataType>::NewInstance()
158 {
159 // ASSERT(m_iTop);
160  if(m_iTop <= 0)
161  {
162  LOG_ERROR("Kein Platz mehr in der Liste!", NULL);
163  }
164  return m_ppFreeObjects[--m_iTop];
165 }

171 
172  //Fehler kontrolle
173 // ASSERT(pInstance >= &(m_pObjectData[0]) &&
174 // pInstance < &(m_pObjectData[m_iNumObject]));
175 // ASSERT(m_iTop < m_iNumObject);
176 
177  m_ppFreeObjects[m_iTop++] = pInstance;
