42// sample hash creation functions
43//---------------------------------------------------------
44inline HASH DRHashRotateLeft( HASH hash, const unsigned int rotateBy )
45{
46 return (hash<<rotateBy)|(hash>>(32-rotateBy));
47}

49//---------------------------------------------------------
50// create a hash from a string
51//---------------------------------------------------------
52inline HASH DRMakeStringHash( const char *pString )
53{
54 HASH ret = 0;
55 char c;
56
57 if( pString )
58 {
59 while( (c=*(pString++)) != '\0' )
60 ret = DRHashRotateLeft(ret,7) + c;
61 }
62 return ret;
63}

66inline HASH DRMakeStringHash(const char *pString, unsigned int str_size)
67{
68 HASH ret = 0;
69
70 if (pString)
71 {
72 for (unsigned int i = 0; i < str_size; i++) {
73 ret = DRHashRotateLeft(ret, 7) + pString[i];
74 }
75 }

79//---------------------------------------------------------
80// create a hash from a filename
81//---------------------------------------------------------
82inline HASH DRMakeFilenameHash( const char *pString )
83{
84 HASH ret = 0;
85 char c;
86
87 if( pString )
88 {
89 while( (c=*(pString++)) != '\0' )
90 {
91 if( c>='A' && c<='Z' )
92 c ^= 32;
93 else if( c=='/' )
94 c = '\\';
95 ret = DRHashRotateLeft(ret,7) + c;
96 }
97 }
98 return ret;
99}

102//einen Hash aus zwei Werten erstellen
103inline DHASH DRMakeDoubleHash(const char* pTypeName, const char* pFileName)
104{
105 HASH hTemp1 = DRMakeStringHash(pTypeName);
106 DHASH hTemp2 = DRMakeFilenameHash(pFileName);
107 hTemp2 = hTemp2 << 16;
108 return ((DHASH)(hTemp1)|(DHASH)(hTemp2));
109}
110//einen Hash aus 3 Int-Werten generiren
111inline DHASH DRMakeThreeIntHash(int i1, int i2, int i3)
112{
113 DHASH h1 = DRHashRotateLeft(i1, 24);
114 h1 = h1 | DRHashRotateLeft(i2, 16);
115 return h1 | i3;
116}
